class Book inherits IO {
    title : Int;
    author : Int;

    initBook(title_p : Int, author_p : Int) : Book {
        {
            title <- title_p;
            author <- author_p;
            self;
        }
    };

    print() : Book {
        {
            out_int(title);
            out_int(author);
            self;
        }
    };

};

class Main inherits IO {
    a: Int <- 8;
    b: Int <- 2;
    main() : Object{
       out_int(a*b)
    };
};
